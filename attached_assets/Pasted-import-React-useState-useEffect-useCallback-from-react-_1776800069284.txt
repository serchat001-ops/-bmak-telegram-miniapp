// ─── Helpers ──────────────────────────────────────────────────────────────────
const ADMIN_EMAILS = ['ben.makoma98@gmail.com', 'serchat001@gmail.com', 'officielbmak@gmail.com'];

function isAdminEmail(email) {
  return ADMIN_EMAILS.includes(email?.toLowerCase());
}

async function hashPassword(password) {
  return bcrypt.hash(password, 10);
}

async function verifyPassword(password, hash) {
  return bcrypt.compare(password, hash);
}

function todayStr() { return new Date().toISOString().split('T')[0]; }

function canCheckinToday(profile) {
  if (!profile?.last_checkin) return true;
  const last = typeof profile.last_checkin === 'string'
    ? profile.last_checkin.slice(0, 10)
    : new Date(profile.last_checkin).toISOString().slice(0, 10);
  return last !== todayStr();
}

function genCode(len = 6) {
  const chars = 'ABCDEFGHJKLMNPQRSTUVWXYZ23456789';
  let code = '';
  for (let i = 0; i < len; i++) code += chars[Math.floor(Math.random() * chars.length)];
  return code;
}

async function cacheProfile(user) {
  if (!user) return;
  await AsyncStorage.setItem(KEY_PROFILE, JSON.stringify(user));
}

// ─── Hook useBmakSession (exporté pour Programs.js) ───────────────────────────
export function useBmakSession() {
  const [bmakBalance, setBmakBalance] = useState(null);
  const [bmakStreak,  setBmakStreak]  = useState(0);

  const syncFromUser = useCallback((user) => {
    if (!user) return;
    if (user.bmak_balance  != null) setBmakBalance(user.bmak_balance);
    if (user.checkin_streak != null) setBmakStreak(user.checkin_streak);
  }, []);

  const restoreSession = useCallback(async () => {
    const cached = await AsyncStorage.getItem(KEY_PROFILE);
    if (!cached) return null;
    try {
      const user = JSON.parse(cached);
      syncFromUser(user);
      // Rafraîchissement silencieux depuis Supabase
      const uid = await AsyncStorage.getItem(KEY_SESSION);
      if (uid) {
        bmakSupabase.from('users').select('*').eq('id', uid).single()
          .then(({ data }) => { if (data) { syncFromUser(data); cacheProfile(data); } })
          .catch(() => {});
      }
      return user;
    } catch { return null; }
  }, [syncFromUser]);

  const fetchBalance = useCallback(async () => {
    const uid = await AsyncStorage.getItem(KEY_SESSION);
    if (!uid) return;
    const { data } = await bmakSupabase.from('users').select('*').eq('id', uid).single();
    if (data) { syncFromUser(data); await cacheProfile(data); }
  }, [syncFromUser]);

  const clearSession = useCallback(async () => {
    await AsyncStorage.multiRemove([KEY_SESSION, KEY_PROFILE, KEY_ADMIN]);
    setBmakBalance(null); setBmakStreak(0);
  }, []);

  return { bmakBalance, bmakStreak, restoreSession, fetchBalance, clearSession, syncFromUser };
}

function LoginScreen({ onSuccess }) {
  const [mode, setMode]         = useState('login');
  const [name, setName]         = useState('');
  const [email, setEmail]       = useState('');
  const [password, setPassword] = useState('');
  const [loading, setLoading]   = useState(false);
  const [showPwd, setShowPwd]   = useState(false);
  const [err, setErr]           = useState('');

  const submit = async () => {
    setErr('');
    if (!email.trim() || !password) { setErr('Email et mot de passe requis.'); return; }
    if (mode === 'register' && !name.trim()) { setErr('Le nom est requis.'); return; }
    setLoading(true);
    try {
      if (mode === 'login') {
        // Étape 1 : récupérer l'utilisateur par email
        const { data: emailCheck, error: emailErr } = await bmakSupabase
          .from('users')
          .select('*')
          .eq('email', email.trim().toLowerCase())
          .single();
        if (emailErr) {
          const code = emailErr?.code;
          if (code === 'PGRST116') {
            setErr('Aucun compte trouvé avec cet email. Veuillez vous inscrire.');
          } else {
            setErr(`Erreur [${code}]: ${emailErr.message}`);
          }
          return;
        }
        if (!emailCheck) { setErr('Aucun compte trouvé avec cet email.'); return; }
        // Étape 2 : vérifier le mot de passe avec bcrypt
        const passwordOk = await verifyPassword(password.trim(), emailCheck.password_hash);
        if (!passwordOk) { setErr('Mot de passe incorrect.'); return; }
        // Étape 3 : session
        await AsyncStorage.setItem(KEY_SESSION, String(emailCheck.id));
        await cacheProfile(emailCheck);
        if (isAdminEmail(emailCheck.email)) await AsyncStorage.setItem(KEY_ADMIN, String(emailCheck.id));
        onSuccess(emailCheck);
      } else {
        // Vérifier si email déjà utilisé
        const { data: existing } = await bmakSupabase
          .from('users').select('id').eq('email', email.trim().toLowerCase()).single();
        if (existing) { setErr('Cet email est déjà utilisé.'); return; }
        // Générer un code de parrainage unique
        let code = genCode();
        const { data: codeExists } = await bmakSupabase
          .from('users').select('id').eq('referral_code', code).single();
        if (codeExists) code = genCode(7);
        // Hacher le mot de passe avec bcrypt
        const hash = await hashPassword(password.trim());
        const { data: newUser, error } = await bmakSupabase
          .from('users')
          .insert({
            email: email.trim().toLowerCase(),
            password_hash: hash,
            display_name: name.trim(),
            auth_type: 'email',
            referral_code: code,
            bmak_balance: 0,
            checkin_streak: 0,
            total_earned: 0,
            total_referrals: 0,
          })
          .select('*')
          .single();
        if (error || !newUser) { setErr(error?.message || 'Erreur lors de l\'inscription.'); return; }
        await AsyncStorage.setItem(KEY_SESSION, String(newUser.id));
        await cacheProfile(newUser);
        onSuccess(newUser);
      }
    } catch (e) {
      setErr(e.message || 'Une erreur est survenue.');
    } finally { setLoading(false); }
  };

  return (
    <ScrollView contentContainerStyle={S.loginContainer} keyboardShouldPersistTaps="handled">
      <View style={S.loginLogo}>
        <Image source={bmakLogo} style={S.loginLogoImg} resizeMode="contain" />
        <Text style={S.loginTitle}>B_MAK</Text>
        <Text style={S.loginSub}>Digital Token • Community & Growth</Text>
      </View>

      <View style={S.tabs}>
        {['login','register'].map(m => (
          <TouchableOpacity key={m} style={[S.tab, mode===m && S.tabActive]} onPress={()=>{setMode(m);setErr('');}}>
            <Text style={[S.tabTxt, mode===m && S.tabTxtActive]}>{m==='login'?'Connexion':'Inscription'}</Text>
          </TouchableOpacity>
        ))}
      </View>

      <View style={S.form}>
        {mode==='register' && (
          <View style={S.inp}>
            <Ionicons name="person-outline" size={18} color="#8b8bba" style={S.inpIcon}/>
            <TextInput style={S.inpField} placeholder="Nom d'affichage" placeholderTextColor="#4a4a7a"
              value={name} onChangeText={setName} autoCapitalize="words"/>
          </View>
        )}
        <View style={S.inp}>
          <Ionicons name="mail-outline" size={18} color="#8b8bba" style={S.inpIcon}/>
          <TextInput style={S.inpField} placeholder="Adresse email" placeholderTextColor="#4a4a7a"
            value={email} onChangeText={setEmail} keyboardType="email-address" autoCapitalize="none"/>
        </View>
        <View style={S.inp}>
          <Ionicons name="lock-closed-outline" size={18} color="#8b8bba" style={S.inpIcon}/>
          <TextInput style={[S.inpField,{flex:1}]} placeholder="Mot de passe" placeholderTextColor="#4a4a7a"
            value={password} onChangeText={setPassword} secureTextEntry={!showPwd}/>
          <TouchableOpacity onPress={()=>setShowPwd(v=>!v)} style={{padding:4}}>
            <Ionicons name={showPwd?'eye-off-outline':'eye-outline'} size={18} color="#8b8bba"/>
          </TouchableOpacity>
        </View>
        {err ? <Text style={S.errTxt}>{err}</Text> : null}
        <TouchableOpacity style={S.submitBtn} onPress={submit} disabled={loading}>
          {loading ? <ActivityIndicator color="#fff"/> :
            <Text style={S.submitTxt}>{mode==='login'?'Se connecter':'Créer un compte'}</Text>}
        </TouchableOpacity>
      </View>
    </ScrollView>
  );
}

// ─── Composant racine ─────────────────────────────────────────────────────────
export default function BmakMining({ onBack }) {
  const [bmakUser, setBmakUser]   = useState(null);
  const [restoring, setRestoring] = useState(true);
  const { bmakBalance, bmakStreak, restoreSession, clearSession, syncFromUser } = useBmakSession();

  useEffect(() => {
    (async () => {
      const user = await restoreSession();
      if (user) setBmakUser(user);
      setRestoring(false);
    })();
  }, []);

  const handleLogout = async () => {
    if (Platform.OS === 'web') {
      // window.confirm fonctionne de manière fiable sur web
      const ok = window.confirm('Voulez-vous vous déconnecter de B_MAK ?');
      if (!ok) return;
      await clearSession();
      setBmakUser(null);
    } else {
      Alert.alert('Déconnexion B_MAK', 'Voulez-vous vous déconnecter ?', [
        { text: 'Annuler', style: 'cancel' },
        { text: 'Déconnecter', style: 'destructive', onPress: async () => {
          await clearSession();
          setBmakUser(null);
        }},
      ]);
    }
  };

  return (
    <SafeAreaView style={S.safe}>
      <StatusBar barStyle="light-content" backgroundColor="#0a0718"/>
      <View style={S.header}>
        {onBack && (
          <TouchableOpacity style={S.backBtn} onPress={onBack}>
            <Ionicons name="arrow-back" size={22} color="#a78bfa"/>
          </TouchableOpacity>
        )}
        <Image source={bmakLogo} style={S.headerLogo} resizeMode="contain"/>
        <Text style={S.headerTitle}>B_MAK Mining</Text>
        <View style={{flex:1}}/>
        {bmakUser && bmakBalance != null && (
          <View style={S.hPill}>
            <Text style={S.hPillTxt}>💎 {Number(bmakBalance).toLocaleString('fr-FR')}</Text>
          </View>
        )}
        {bmakUser && bmakStreak > 0 && (
          <View style={[S.hPill,{backgroundColor:'#431407'}]}>
            <Text style={[S.hPillTxt,{color:'#fb923c'}]}>🔥 {bmakStreak}</Text>
          </View>
        )}
      </View>

      {!bmakConfigured ? (
        <View style={S.center}>
          <Image source={bmakLogo} style={{width:80,height:80,marginBottom:16}} resizeMode="contain"/>
          <Ionicons name="cloud-offline-outline" size={40} color="#f87171" style={{marginBottom:8}}/>
          <Text style={[S.restoringTxt,{color:'#f87171',fontSize:15,fontWeight:'700',textAlign:'center'}]}>
            Configuration manquante
          </Text>
          <Text style={[S.restoringTxt,{textAlign:'center',paddingHorizontal:32,marginTop:8}]}>
            Ajoutez ces 2 variables dans les Secrets Replit :{'\n\n'}
            EXPO_PUBLIC_BMAK_SUPABASE_URL{'\n'}
            EXPO_PUBLIC_BMAK_SUPABASE_ANON_KEY
          </Text>
        </View>
      ) : restoring ? (
        <View style={S.center}>
          <Image source={bmakLogo} style={{width:90,height:90,marginBottom:18}} resizeMode="contain"/>
          <ActivityIndicator color="#a78bfa" size="large"/>
          <Text style={S.restoringTxt}>Chargement…</Text>
        </View>
      ) : bmakUser ? (
        <Dashboard initUser={bmakUser} onLogout={handleLogout} syncFromUser={syncFromUser}/>
      ) : (
        <LoginScreen onSuccess={u => { setBmakUser(u); syncFromUser(u); }}/>
      )}
    </SafeAreaView>
  );
}
